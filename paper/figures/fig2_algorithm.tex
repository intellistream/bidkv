\begin{figure}[t]
\begin{algorithm}[H]
\caption{Utility-Ranked Victim Ordering}\label{alg:greedy}
\begin{algorithmic}[1]
\Require Bid pool $\mathcal{P} = \{b_1, b_2, \ldots, b_B\}$
\Ensure Utility-ranked ordering $\mathcal{O}$
\ForAll{$b \in \mathcal{P}$}
  \State $U(b) \gets \dfrac{b.r}{b.\delta + \varepsilon}$
  \Comment{reclamation utility, $\varepsilon = 10^{-3}$}
\EndFor
\State Sort $\mathcal{P}$ by $U(b)$ in decreasing order $\to \mathcal{O}$
\State \Return $\mathcal{O}$
  \Comment{cached; adapter pops top entry at each pressure event}
\end{algorithmic}
\end{algorithm}
